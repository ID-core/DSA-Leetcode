// 876. Middle of the Linked List
// https://leetcode.com/problems/middle-of-the-linked-list/description/

class Solution {
    public ListNode middleNode(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;
        while(fast!= null && fast.next!=null)
        {
            slow=slow.next;
            fast=fast.next.next;
        }
        return slow;
    }
}



// 206. Reverse Linked List
// https://leetcode.com/problems/reverse-linked-list/description/

class Solution {
    public ListNode reverseList(ListNode head) {
        ListNode prev = null;
        ListNode current = head;
        while(current != null)
        {
        ListNode next= current.next;
        current.next = prev;
        prev = current;
        current = next;
        }
        return prev; //new head
    }
}



// 234. Palindrome Linked List
// https://leetcode.com/problems/palindrome-linked-list/description/

class Solution {
    public boolean isPalindrome(ListNode head) {
        ArrayList<Integer> list = new ArrayList<>();
        while (head != null) {
            list.add(head.val);
            head = head.next;
        }
        int right = list.size() - 1;
        for (int left = 0; left < right; left++) {

            if (!list.get(left).equals(list.get(right))) {
                return false;
            }
            right--;
        }
        return true;
    }
}



// 92. Reverse Linked List II
// https://leetcode.com/problems/reverse-linked-list-ii/description/

class Solution {
    public ListNode reverseBetween(ListNode head, int left, int right) {

        if (head == null || left == right) return head;

        ListNode dummy = new ListNode(0);
        dummy.next = head;

        // prev will point to (left - 1)th node
        ListNode prev = dummy;
        for (int i = 1; i < left; i++) {
            prev = prev.next;
        }

        // current starts at left
        ListNode current = prev.next;

        // reverse between left and right
        for (int i = 0; i < right - left; i++) {
            ListNode next = current.next;
            current.next = next.next;
            next.next = prev.next; //forces the extracted node to point to front node
            prev.next = next;
        }

        return dummy.next;
    }
}



// 21. Merge two Sorted lists
// https://leetcode.com/problems/merge-two-sorted-lists/description/

class Solution {
    public ListNode mergeTwoLists(ListNode list1, ListNode list2) {
        if (list1 == null)
            return list2;
        if (list2 == null)
            return list1;
        ListNode temp;

        if (list1.val < list2.val) {
            list1.next = mergeTwoLists(list1.next, list2);
            return list1;
        } else {
            list2.next = mergeTwoLists(list1, list2.next);
            return list2;
        }
    }
}



// 141. Linked List Cycle
// https://leetcode.com/problems/linked-list-cycle/

public class Solution {
    public boolean hasCycle(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;
        if (head == null || head.next == null) //base case for empty list or single node
            return false;

        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;

            if (slow == fast)
                return true;
        }
        return false;
    }
}



// 203. Remove Linked List Elements
// https://leetcode.com/problems/remove-linked-list-elements/

class Solution {
    public ListNode removeElements(ListNode head, int val) {
        ListNode dummy = new ListNode(0); //creating a listnode named dummy at index 0
        dummy.next = head;

        ListNode current = dummy;
        while (current.next != null) {
            if (current.next.val == val) {
                current.next = current.next.next;
            } else {
                current = current.next;
            }
        }
        return dummy.next;
    }
}



// 143. Reorder List
// https://leetcode.com/problems/reorder-list/description/

class Solution {
    public void reorderList(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;
        while (fast.next != null && fast.next.next != null) {
            slow = slow.next;
            fast = fast.next.next;
        }
        ListNode second = reverselist(slow.next); // splitting second half
        slow.next = null; //splitting the list

        ListNode first = head; //first node starts from head

        while (second != null) {
            ListNode temp1 = first.next; // saving values before breaking out there nodes
            ListNode temp2 = second.next;

            first.next = second; // connect first node with the second
            second.next = temp1; // connecting second node with the first one

            first = temp1;
            second = temp2;
        }

    }

    private ListNode reverselist(ListNode head) {
        ListNode prev = null;
        while (head != null) {
            ListNode next = head.next;
            head.next = prev;
            prev = head;
            head = next;
        }
        return prev;
    }
}



// 19. Remove Nth Node From End of List
// https://leetcode.com/problems/remove-nth-node-from-end-of-list/description/

class Solution {
    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode dummy = new ListNode(0); // dummy node to delete the necessary node
        ListNode slow = dummy;
        ListNode fast = dummy;
        dummy.next = head; //connecting dummy to head

        for(int i=0; i<n;i++){
            fast=fast.next; // creating gap between slow and fast of 'n' so that when fast will be at last node. slow will be automatically at the desired location
        }

        while(fast.next!=null)
        {
            slow=slow.next;
            fast=fast.next;
        }
        slow.next = slow.next.next;

    return dummy.next;
    }
}



// 61. Rotate List
// https://leetcode.com/problems/rotate-list/?envType=problem-list-v2&envId=linked-list

class Solution {
    public ListNode rotateRight(ListNode head, int k) {
        ListNode left = head;
        ListNode right = head;

        int length = 0;
        ListNode temp = head;
        if (head == null || head.next == null || k == 0) {
            return head;
        }

        while (temp != null) {
            length++;
            temp = temp.next; //counting length of list
        }

        k = k % length;
        if (k == 0)
            return head; //reducing the number of rotation if no of rotation is > length

        for (int i = 0; i < k; i++) {
            right = right.next;
        }
        while (right.next != null) // making left on thee node from where we have to split
        {
            left = left.next;
            right = right.next;
        }

        ListNode new_node = left.next;
        left.next = null;
        right.next = head;
        return new_node;
    }
}



// 82. Remove Duplicates from Sorted List II
// https://leetcode.com/problems/remove-duplicates-from-sorted-list-ii/?envType=problem-list-v2&envId=linked-list

class Solution {
    public ListNode deleteDuplicates(ListNode head) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode prev = dummy; //to keep track of the last node we added
        ListNode curr = head;
        while (curr != null) {
            if (curr.next != null && curr.val == curr.next.val) //detecting if net is duplicate
            {
                int dupl = curr.val;
                while (curr != null && curr.val == dupl) //checking for all duplicates
                {
                    curr = curr.next;
                }
                prev.next = curr; //removing duplicate
            } else {
                prev = curr;
                curr = curr.next;
            }
        }
        return dummy.next;
    }
}
