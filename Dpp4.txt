// 191. Number of 1 Bits
// https://leetcode.com/problems/number-of-1-bits/description/


class Solution {
    public int hammingWeight(int n) {
    int count=0;
    String binaryString = Integer.toBinaryString(n);
    for(int i=0;i<binaryString.length();i++)
    {
        if(binaryString.charAt(i)=='1')
        {
            count++;
        }
    }
    return count;
    }
}



// 338. Counting Bits
// https://leetcode.com/problems/counting-bits/description/


class Solution {
    public int[] countBits(int n) {
        int[] ans = new int[n+1];
        int count=0;
        for(int a=0;a <=n;a++)
        {
            String binaryString = Integer.toBinaryString(a);
            for(int i=0;i<binaryString.length();i++)
            {
                if(binaryString.charAt(i)=='1')
                {
                    count++;
                }
            }
        ans[a] = count;
        count=0;
        }
        return ans;
    }
}



// 268. Missing Number
// https://leetcode.com/problems/missing-number/


//approach :Sum of first n natural numbers = n(n+1)/2. Now if we subtract the sum of given array elements, the remainder must be the missing number.
class Solution {
    public int missingNumber(int[] nums) {
        int array_sum=0;
        int actual_sum = nums.length * (nums.length+1)/2;
        for(int i=0; i<nums.length; i++)
        {
            array_sum += nums[i];
        }
        return actual_sum- array_sum;
    }
}
