// Q1. Set Mismatch
// https://leetcode.com/problems/set-mismatch/description/?envType=problem-list-v2&envId=dsa-linear-shoal-array-ii

class Solution {
    public int[] findErrorNums(int[] nums) {
        int n=nums.length;
        int duplicate=-1;
        int missing =-1;
        int[] counts = new int[n+1];
        for(int i =0; i<n ;i++)
        {
           counts[nums[i]]++;
        }
        for(int i=0;i<=n;i++)
        {
            if(counts[i]==2)
                duplicate=i;
            if(counts[i]==0)
                missing=i;

        }
        return new int[]{duplicate,missing};
    }
}


// Q2. How many numbers are smaller than the current number
// https://leetcode.com/problems/how-many-numbers-are-smaller-than-the-current-number/description/?envType=problem-list-v2&envId=dsa-linear-shoal-array-ii

class Solution {
    public int[] smallerNumbersThanCurrent(int[] nums) {
        int n =nums.length;
        int count=0;
        int[] ans= new int[n];
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(nums[j]<nums[i])
                count++;
            }
            ans[i]= count;
            count=0;
        }
         return ans;
}}


// Q3. Find all numbers disappeared in an Array
// https://leetcode.com/problems/find-all-numbers-disappeared-in-an-array/?envType=problem-list-v2&envId=dsa-linear-shoal-array-ii

class Solution {
    public List<Integer> findDisappearedNumbers(int[] nums) {
        int n = nums.length;
        boolean[] seen = new boolean[n + 1];
        List<Integer> ans = new ArrayList<>();

        for (int num : nums) {
            seen[num] = true;
        }

        for (int i = 1; i <= n; i++) {
            if (!seen[i]) ans.add(i);
        }

        return ans;
    }
}
