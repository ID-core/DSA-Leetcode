// 344. Reverse String
// https://leetcode.com/problems/reverse-string/?envType=problem-list-v2&envId=string

class Solution {
    public void reverseString(char[] s) {
        char temp; int n=s.length;
        for(int i=0;i<n/2;i++)
        {
            temp = s[i];
            s[i] = s[n-i-1];
            s[n-i-1] = temp;
        }
    }
}
